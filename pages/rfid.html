#include <WiFi.h> #include <WebSocketsServer.h> // Library: "WebSockets" by Markus Sattler #include <SPI.h> #include <MFRC522.h> // Library: "MFRC522" by GithubCommunity // ── CHANGE THESE ────────────────────────────────────── #define WIFI_SSID "YOUR_WIFI_NAME" #define WIFI_PASSWORD "YOUR_WIFI_PASSWORD" // ── PINS (NodeMCU-32S default SPI) ─────────────────── #define SS_PIN 5 // SDA / CS #define RST_PIN 22 // ───────────────────────────────────────────────────── MFRC522 rfid(SS_PIN, RST_PIN); WebSocketsServer wsServer(81); String lastUID = ""; unsigned long lastScanTime = 0; const unsigned long SCAN_COOLDOWN = 1000; // ms between same-card reads void webSocketEvent(uint8_t num, WStype_t type, uint8_t* payload, size_t length) { if (type == WStype_CONNECTED) { Serial.printf("[WS] Client #%u connected\n", num); wsServer.sendTXT(num, "{\"status\":\"ESP32_CONNECTED\"}"); } else if (type == WStype_DISCONNECTED) { Serial.printf("[WS] Client #%u disconnected\n", num); } } void setup() { Serial.begin(115200); SPI.begin(); rfid.PCD_Init(); rfid.PCD_DumpVersionToSerial(); Serial.print("[WiFi] Connecting to "); Serial.println(WIFI_SSID); WiFi.begin(WIFI_SSID, WIFI_PASSWORD); while (WiFi.status() != WL_CONNECTED) { delay(500); Serial.print("."); } Serial.println(); Serial.print("[WiFi] Connected! IP: "); Serial.println(WiFi.localIP()); wsServer.begin(); wsServer.onEvent(webSocketEvent); Serial.println("[WS] WebSocket server started on port 81"); Serial.println("[RFID] Ready — scan a card..."); } void loop() { wsServer.loop(); // Check for new card if (!rfid.PICC_IsNewCardPresent() || !rfid.PICC_ReadCardSerial()) return; // Build UID string String uid = ""; for (byte i = 0; i < rfid.uid.size; i++) { if (i > 0) uid += " "; if (rfid.uid.uidByte[i] < 0x10) uid += "0"; uid += String(rfid.uid.uidByte[i], HEX); } uid.toUpperCase(); // Cooldown to avoid duplicate sends unsigned long now = millis(); if (uid == lastUID && (now - lastScanTime) < SCAN_COOLDOWN) { rfid.PICC_HaltA(); rfid.PCD_StopCrypto1(); return; } lastUID = uid; lastScanTime = now; // Build JSON and broadcast String msg = "{\"uid\":\"" + uid + "\"}"; wsServer.broadcastTXT(msg); Serial.print("[SCAN] UID: "); Serial.println(uid); rfid.PICC_HaltA(); rfid.PCD_StopCrypto1(); delay(500); }
